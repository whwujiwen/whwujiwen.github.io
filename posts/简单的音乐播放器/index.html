基于Vue的音乐播放器

爱好

一、主要的接口

https://autumnfish.cn/search?keywords="有多少爱可以重来"		//根据关键字来获取歌曲列表

https://autumnfish.cn/song/url?id=152556					  //根据歌曲的id来获取歌曲的播放地址

二 、完整的代码实现

<template>
    <div class="music">
        <el-header style="background: #545c64;border-radius: 10px">

            <el-menu
                style=""
                :default-active="activeIndex2"
                class="el-menu-demo"
                mode="horizontal"
                @select="handleSelect"
                background-color="#545c64"
                text-color="#fff"
                active-text-color="#ffd04b">
                <el-menu-item><a style="font-size:x-large;font-family:Microsoft YaHei;color:#7AD038;padding-right:100px;">Bolg</a></el-menu-item>
                <el-menu-item index="1"><router-link tag="a" target="_blank" style="width: 100%;height: 100%" to="/index">首页</router-link></el-menu-item>
                <el-menu-item><el-divider direction="vertical"></el-divider></el-menu-item>
                <el-menu-item index="2" ><router-link tag="a" target="_blank" style="width: 100%;height: 100%" to="/classify">分类</router-link></el-menu-item>
                <el-menu-item><el-divider direction="vertical"></el-divider></el-menu-item>
                <el-menu-item index="3"><router-link style="width: 100%;height: 100%" to="/music">音乐欣赏</router-link></el-menu-item>
                <el-menu-item><el-divider direction="vertical"></el-divider></el-menu-item>
                <el-menu-item index="4"><router-link tag="a" target="_blank" style="width: 100%;height: 100%" to="/about">关于我</router-link></el-menu-item>
                <el-menu-item><el-divider direction="vertical"></el-divider></el-menu-item>
                <el-menu-item index="5"><router-link tag="a" target="_blank" style="width: 100%;height: 100%" to="/login">后台管理</router-link></el-menu-item>
            </el-menu>
        </el-header>

        <div style="padding:30px;max-width: 60%;margin:30px auto;border-radius:4px;box-shadow:0 2px 4px rgba(0,0,0,.12), 0 0 6px rgba(0,0,0,.04);">
            <div style="width: 100%;text-align: center"><h1>轻松一刻</h1></div>
            <el-divider></el-divider>

            <span style="color: #66B1FF">搜索:</span>
            <el-input
                style="margin-top:15px;"
                placeholder="请输入关键字..."
                v-model="input">
            </el-input>
            <div style="width:100%">
                <el-button type="primary" icon="el-icon-search" style="width: 100%;margin:15px auto;" @click="search">搜索</el-button>
            </div>
            <el-divider></el-divider>

            <!--歌曲列表-->
            <el-table
                :data="musics"
                style="width: 100%;font-size: larger">
                <el-table-column
                    label="歌名"
                    width="300">
                    <template slot-scope="scope">
                        <span style="margin-left: 10px">{{ scope.row.name }}</span>
                    </template>
                </el-table-column>

                <el-table-column
                    label="歌手"
                    width="300">
                    <template slot-scope="scope">
                        <el-popover trigger="hover" placement="top">
                            <p>歌名: {{ scope.row.artists[0].name }}</p>
                            <div slot="reference" class="name-wrapper">
                                <el-tag size="medium">{{ scope.row.artists[0].name }}</el-tag>
                            </div>
                        </el-popover>
                    </template>
                </el-table-column>
                <el-table-column
                    label="歌名"
                    width="300">
                    <template slot-scope="scope">
                        <span style="margin-left: 10px"><span v-if="current==scope.row.id">正在播放<i class="el-icon-loading
"></i></span></span>
                    </template>
                </el-table-column>
                <el-table-column label="操作">
                    <template slot-scope="scope">
                        <el-button
                            size="success"
                            @click="playMusic(scope.row.id)">播放</el-button>
                        <el-button
                            type="info"
                            @click="stopMusic(scope.row.id)">停止</el-button>
                    </template>
                </el-table-column>
            </el-table>
            <footer class="footer" style="display: none">
                <audio :src="url" loop autoplay controls></audio>
            </footer>
        </div>
    </div>
</template>

<script>
    export default {
        name: "Music",
        data() {
            return {
                current:"0",
                input: '',
                musics:[],
                url:"",
                id:""
            };
        },
        methods: {
            stopMusic(id){
                this.url="";
                this.current=""
            },
            search(){
                let _this = this
                this.$http.get("https://autumnfish.cn/search?keywords="+_this.input).then((res)=>{
                    console.log(res.data.result.songs);
                    _this.musics = res.data.result.songs
                });
            },
            playMusic(id){
                this.id = id
                this.current=id
                let _this = this
                this.$http.get("https://autumnfish.cn/song/url?id="+id).then((res)=>{
                    console.log(res.data.data[0].url);
                    this.url = res.data.data[0].url
                });
            }
        }
    }
</script>
